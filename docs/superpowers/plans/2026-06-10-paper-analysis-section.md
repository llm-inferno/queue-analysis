```python
"""One-shot /target sweep over M ∈ [1, 256] for the baseline scenario.

Caches RPSTargetTTFT, RPSTargetITL, Throughput per M to
paper/data/baseline_lambda_sweep.json so figure scripts run without
hitting the analyzer.

Requires the queue-analysis Go server running on :8080.
"""
import json
import sys
from pathlib import Path

import requests

REPO = Path(__file__).resolve().parents[2]
SCENARIOS = json.loads((REPO / "nous" / "scenarios.json").read_text())
OUT = REPO / "paper" / "data" / "baseline_lambda_sweep.json"
URL = "http://localhost:8080/target"
M_MIN, M_MAX = 1, 256


def main() -> int:
    baseline = next(s for s in SCENARIOS if s["name"] == "baseline")
    base_payload = {
        "AvgInputTokens": baseline["AvgInputTokens"],
        "AvgOutputTokens": baseline["AvgOutputTokens"],
        "alpha": baseline["alpha"],
        "beta": baseline["beta"],
        "gamma": baseline["gamma"],
        "MaxQueueSize": baseline["MaxQueueSize"],
        "TargetTTFT": baseline["TargetTTFT"],
        "TargetITL": baseline["TargetITL"],
    }

    rows = []
    for m in range(M_MIN, M_MAX + 1):
        payload = dict(base_payload, MaxBatchSize=m)
        r = requests.post(URL, json=payload, timeout=30)
        if r.status_code == 400:
            rows.append({"m": m, "infeasible": True})
            continue
        r.raise_for_status()
        body = r.json()
        rows.append({
            "m": m,
            "infeasible": False,
            "throughput": body.get("Throughput"),
            "RPSTargetTTFT": body.get("RPSTargetTTFT"),
            "RPSTargetITL": body.get("RPSTargetITL"),
        })
        if m % 32 == 0:
            print(f"  M={m}: λ*_TTFT={body.get('RPSTargetTTFT'):.4f}  "
                  f"λ*_ITL={body.get('RPSTargetITL'):.4f}  f={body.get('Throughput'):.4f}")

    OUT.parent.mkdir(parents=True, exist_ok=True)
    OUT.write_text(json.dumps({"scenario": "baseline", "rows": rows}, indent=2))
    print(f"\nWrote {len(rows)} rows to {OUT}")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

**Note on field names:** if your inspection in Step 2 reveals the response uses different names (e.g. lowercase `throughput`), adjust `body.get(...)` accordingly. Do NOT silently fall back — if a field is missing, raise.

```python
def fig4_R_crossover():
    sweep = load_baseline_sweep()
    rows = [r for r in sweep["rows"] if not r["infeasible"]]
    truth = load_truth("baseline")
    M_star = truth["M_truth"]

    ms = np.array([r["m"] for r in rows])
    lam_ttft = np.array([r["RPSTargetTTFT"] for r in rows])
    lam_itl = np.array([r["RPSTargetITL"] for r in rows])
    R = lam_ttft / lam_itl
    f_min = np.minimum(lam_ttft, lam_itl)

    # Limit R panel to rising phase only (caption notes non-monotone past M*).
    rising = ms <= M_star + 5

    fig, (ax_top, ax_bot) = plt.subplots(2, 1, figsize=(5.8, 4.2),
                                          sharex=True,
                                          gridspec_kw={"height_ratios": [3, 2]})
    ax_top.plot(ms, f_min, color="C0")
    ax_top.axvline(M_star, color="gray", linestyle="--", linewidth=1)
    ax_top.set_ylabel("$f(M)$ [RPS]")
    ax_top.set_ylim(bottom=0)

    ax_bot.plot(ms[rising], R[rising], color="C3")
    ax_bot.axhline(1.0, color="gray", linestyle=":", linewidth=1)
    ax_bot.axvline(M_star, color="gray", linestyle="--", linewidth=1)
    ax_bot.scatter([M_star], [R[ms == M_star][0]], color="C3", zorder=5)
    ax_bot.set_xlabel("MaxBatchSize $M$")
    ax_bot.set_ylabel(r"$R(M)=\lambda^*_{\mathrm{TTFT}}/\lambda^*_{\mathrm{ITL}}$")
    ax_bot.set_xlim(0, M_star + 5)
    fig.savefig(FIG_DIR / "fig4_R_crossover.pdf")
```

```tex
\subsection{The plateau and the no-crossover case}
\label{sec:plateau}

\paragraph{Structural plateau.} For $M \geq M^*$, the binding constraint
switches from TTFT to ITL, and $\lambda^*_{\mathrm{ITL}}(M)$ saturates: the
additional batch slots cannot be used productively given the ITL target. So
$f(M)$ is bounded above and approaches a limit as $M$ grows.

\paragraph{Numerical observation.} Across our four scenarios, $f$ varies
within $\sim\!10^{-4}$ of $f^* := f(M^*)$ over the plateau. This is \emph{not}
evidence of true flatness at machine precision: the analyzer's iterative
solver introduces noise at that scale, and the structural claim only
guarantees boundedness, not constancy. We therefore treat any
$M \geq M^*$ as achieving $f^*$ up to numerical precision and acknowledge a
small lower bound on achievable gap that is set by the solver's noise floor.

\paragraph{No-crossover.} Define
\begin{equation}
  \mathrm{wait\_budget} \;=\; \mathrm{TargetTTFT} - \mathrm{TargetITL}
                              - \mathrm{PrefillTime}(n_{\mathrm{ITL}}).
\end{equation}
When $\mathrm{wait\_budget} < 0$, TTFT never becomes the binding constraint
on the feasible $M$ range: $M^*$ effectively sits at or beyond $M_{\max}$, and
any large-enough $M$ achieves $f^*$ (within solver noise). This is the
trivial extreme of the plateau argument. The classifier
$\mathrm{sign}(\mathrm{wait\_budget})$ distinguishes the two cases at zero
oracle cost.
```
